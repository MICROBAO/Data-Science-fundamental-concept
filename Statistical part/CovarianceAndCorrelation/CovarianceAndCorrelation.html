Bi-variate Normal distributions and Covariance matrices -- Revisited¶

In [14]: 
%%R -w 600 -h 600

data(cars)

X = as.matrix(cars)
means  = apply(X, 2, mean)
sigmas = apply(X, 2, sd)
n = dim(C)[1]
p = dim(C)[2]

Sigma = cov(X)  # covariance matrix
detSigma = det(Sigma)

W = solve(Sigma)  #  = inverse(Sigma)

gaussian = function(xvec) {
     1 / sqrt(2*pi)^2 / sqrt(detSigma) *
         exp( -1/2 * ( t(xvec-means) %*% W %*% (xvec-means) )[1,1] )
}

# generate a surface, filled with heights of a 2D Gaussian

nvalues = 61
xvalues = seq(-3,3, length=nvalues) * sigmas[1] + means[1]
yvalues = seq(-3,3, length=nvalues) * sigmas[2] + means[2]

surface = matrix(0, nvalues, nvalues)
for (i in 1:nvalues) {
   for (j in 1:nvalues) {
      surface[i,j] = gaussian( c(xvalues[i],yvalues[j]) )
   }
}

# plot the 2D Gaussian ("bi-variate normal distribution")

persp(surface, x=xvalues, y=yvalues, main="2D normal distribution")

contour(surface, x=xvalues, y=yvalues, main="the same 2D normal distribution")

}

Interpreting the Eigendecomposition Geometrically¶

In [19]: 
%%R -h 400 -w 800

LinearTransformPlot = function (A = matrix(c(9/5,2/5,2/5,3/5),2,2), n = 60) {

#   The 2x2 matrix A should be real and symmetric;
#   it is used to map the unit circle (with a total of n points) to an ellipse;
#   the axes of the ellipse are the eigenvectors of A, stretched by its eigenvalues.

   t = seq(0, 2*pi, length = n)

   x = rbind( cos(t) , sin(t) )   #   x  =  a sequence of n vectors (points on the unit circle)
   Ax = A %*% x;                  #  Ax  =  the result of transforming all vectors in x by A

   x1 = x[1,];
   x2 = x[2,];

   Ax1 = Ax[1,];
   Ax2 = Ax[2,];

   A_eigen = eigen(A)
   lambda = A_eigen$values
   L      = diag(A_eigen$values)
   Q      = A_eigen$vectors

   e1 = Q[,1]  #  1st eigenvector
   e2 = Q[,2]  #  2nd eigenvector

   if (max(e1) <= 0) {
      e1 = -e1; Q[,1] = -Q[,1];  # if e is an eigenvector, then -e is an eigenvector too:
      e2 = -e2; Q[,2] = -Q[,2];  # so we choose e1 so that it has at least one positive entry.
   }

   xmax = max( lambda[1], 1.0 ) * 1.33
   ymax = max( lambda[2], 1.0 ) * 1.33

   QL = Q %*% L
   Id = diag(c(1,1))
   zeroes = c(0,0)

   row_format = " ( % 4.2f  % 4.2f ) "
   A_format = paste(" A\n ", row_format, "\n", row_format)
   A_values = sprintf( A_format, A[1,1],A[1,2], A[2,1],A[2,2] )
   QL_format = paste(" Q                       L\n ", row_format, row_format, "\n", row_format, row_format)
   QL_values = sprintf(QL_format, Q[1,1],Q[1,2],L[1,1],L[1,2], Q[2,1],Q[2,2],L[2,1],L[2,2])

   opar = par(mfrow=c(1,2), cex=0.85, cex.main=1.0, cex.sub=1.0, font.sub=2)  # make a 1x2 set of plots in one figure

   plot( x1, x2, col="red", pch=20,
      asp=1.0, xlim = c(-xmax, xmax), ylim = c(-ymax,ymax), xlab="", ylab="",
      main="transformation by A of vectors on the unit circle", sub=A_values, col.sub="blue" )
   arrows( x1, x2, Ax1, Ax2, col="green", lwd=1.0, angle=8, length=0.1 )
   arrows( zeroes, zeroes,  Q[1,],  Q[2,], col="red",  lwd=2, angle=8, length=0.1 )
   arrows( zeroes, zeroes, QL[1,], QL[2,], col="blue", lwd=2, angle=8, length=0.1 )
   points( Ax1, Ax2, col="blue", pch=20)

   plot( x1, x2, col="red", pch=20,
      asp=1.0, xlim = c(-xmax, xmax), ylim = c(-ymax,ymax), xlab="", ylab="",
      main="explanation of the transformation by A", sub=QL_values, col.sub="blue" )
   points( Ax1, Ax2, col="blue", pch=20)
